## Gap Properties

### gap

Set gap between grid items (with spacing scale):

```jsx
<Box display="grid" gap={2}>
  <Box>Item 1</Box>
  <Box>Item 2</Box>
  <Box>Item 3</Box>
</Box>
```

```javascript

      gap: getValue(transformer, propValue),
```

## Template Properties

### gridTemplateColumns

Define column structure:

```jsx
{/* Three equal columns */}
<Box display="grid" gridTemplateColumns="1fr 1fr 1fr">
  <Box>Column 1</Box>
  <Box>Column 2</Box>
  <Box>Column 3</Box>
</Box>

{/* Using repeat */}
<Box display="grid" gridTemplateColumns="repeat(3, 1fr)" gap={2}>
  <Box>1</Box>
  <Box>2</Box>
  <Box>3</Box>
</Box>

{/* Fixed and flexible columns */}
<Box display="grid" gridTemplateColumns="200px 1fr 200px">
  <Box>Sidebar</Box>
  <Box>Main</Box>
  <Box>Sidebar</Box>
</Box>

{/* Auto-fit responsive columns */}
<Box display="grid" gridTemplateColumns="repeat(auto-fit, minmax(250px, 1fr))" gap={2}>
  <Box>Auto-sized column</Box>
  <Box>Auto-sized column</Box>
  <Box>Auto-sized column</Box>
</Box>
```

## Common Patterns

### Basic Grid Layout

```jsx
<Box
  display="grid"
  gridTemplateColumns="repeat(3, 1fr)"
  gap={2}
>
  <Box>1</Box>
  <Box>2</Box>
  <Box>3</Box>
  <Box>4</Box>
  <Box>5</Box>
  <Box>6</Box>
</Box>
```

### Responsive Grid

```jsx
<Box
  display="grid"
  gridTemplateColumns={{
    xs: '1fr',
    sm: 'repeat(2, 1fr)',
    md: 'repeat(3, 1fr)',
    lg: 'repeat(4, 1fr)',
  }}
  gap={{ xs: 1, sm: 2, md: 3 }}
>
  <Box>Item 1</Box>
  <Box>Item 2</Box>
  <Box>Item 3</Box>
  <Box>Item 4</Box>
</Box>
```

### Card Grid

```jsx
<Box
  display="grid"
  gridTemplateColumns="repeat(auto-fill, minmax(300px, 1fr))"
  gap={3}
>
  {cards.map((card) => (
    <Box
      key={card.id}
      border={1}
      borderColor="divider"
      borderRadius={2}
      p={2}
    >
      {card.content}
    </Box>
  ))}
</Box>
```

### Dashboard Layout

```jsx
<Box
```

### Image Gallery

```jsx

  display="grid"
  gridTemplateColumns="repeat(auto-fit, minmax(200px, 1fr))"
```

## TypeScript

```tsx
import { Box } from '@mui/system';

interface GridContainerProps {
  columns?: number | string;
  gap?: number;
  children: React.ReactNode;
}

function GridContainer({ columns = 3, gap = 2, children }: GridContainerProps) {
  return (
    <Box
      display="grid"
      gridTemplateColumns={
        typeof columns === 'number' ? `repeat(${columns}, 1fr)` : columns
      }
      gap={gap}
    >
      {children}
    </Box>
  );
}
```
