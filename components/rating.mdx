---
title: Rating
description: Material UI Rating component displays a rating input for collecting user feedback.
---

Ratings provide insight into others' opinions and experiences with a product. Users can also rate products they've purchased.

## Import

```tsx
import Rating from '@mui/material/Rating';
```

## Basic Usage

```tsx
import * as React from 'react';
import Box from '@mui/material/Box';
import Rating from '@mui/material/Rating';
import Typography from '@mui/material/Typography';

export default function BasicRating() {
  const [value, setValue] = React.useState<number | null>(2);

  return (
    <Box sx={{ '& > legend': { mt: 2 } }}>
      <Typography component="legend">Controlled</Typography>
      <Rating
        name="simple-controlled"
        value={value}
        onChange={(event, newValue) => {
          setValue(newValue);
        }}
      />
      <Typography component="legend">Read only</Typography>
      <Rating name="read-only" value={value} readOnly />
      <Typography component="legend">Disabled</Typography>
      <Rating name="disabled" value={value} disabled />
    </Box>
  );
}
```

## Props

<ParamField path="value" type="number | null">
  The rating value.
</ParamField>

<ParamField path="defaultValue" type="number" default="null">
  The default value. Use when the component is not controlled.
</ParamField>

<ParamField path="max" type="number" default="5">
  Maximum rating.
</ParamField>

<ParamField path="precision" type="number" default="1">
  The minimum increment value change allowed.
</ParamField>

<ParamField path="size" type="'small' | 'medium' | 'large'" default="'medium'">
  The size of the component.
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  If `true`, the component is disabled.
</ParamField>

<ParamField path="readOnly" type="boolean" default="false">
  Removes all hover effects and pointer events.
</ParamField>

<ParamField path="highlightSelectedOnly" type="boolean" default="false">
  If `true`, only the selected icon will be highlighted.
</ParamField>

<ParamField path="icon" type="ReactNode">
  The icon to display. Defaults to `<Star fontSize="inherit" />`.
</ParamField>

<ParamField path="emptyIcon" type="ReactNode">
  The icon to display when empty. Defaults to `<StarBorder fontSize="inherit" />`.
</ParamField>

<ParamField path="emptyLabelText" type="ReactNode" default="'Empty'">
  The label read when the rating input is empty.
</ParamField>

<ParamField path="getLabelText" type="(value: number) => string">
  Accepts a function which returns a string value that provides a user-friendly name for the current value of the rating. This is important for screen reader users. Default returns `${value || '0'} Star${value !== 1 ? 's' : ''}`.
</ParamField>

<ParamField path="onChange" type="(event: React.SyntheticEvent, value: number | null) => void">
  Callback fired when the value changes.
</ParamField>

<ParamField path="onChangeActive" type="(event: React.SyntheticEvent, value: number) => void">
  Callback function that is fired when the hover state changes.
</ParamField>

<ParamField path="name" type="string">
  The name attribute of the radio `input` elements. This input `name` should be unique within the page.
</ParamField>

<ParamField path="sx" type="SxProps<Theme>">
  The system prop that allows defining system overrides as well as additional CSS styles.
</ParamField>
